%authblk {{{

%Redefines the \author command to work as normal or to allow a footnote style of author/affiliation input.
\usepackage{authblk} 

%}}}

%}}}
%Settings {{{

%Font style
\renewcommand\Affilfont{\itshape\small}

%}}}
%Examples {{{

%1 {{{

\documentclass{report}

\usepackage{authblk} 

\author[1]{AuthorName1}
\author[2]{AuthorName2}
\author[3]{AuthorName3}

\affil[1]{Institute 1, Address 1} 
\affil[2]{Institute 2, Address 2}
\affil[3]{Institute 3, Address 3}

\title{} 

\begin{document}
	\maketitle 
	Goodbye World!
\end{document}

%}}}
