\definecolor{keycolor}{RGB}{172, 42, 42}
\definecolor{mbleu}{RGB}{64,96,127}
\definecolor{vimvert}{RGB}{46, 139, 87}

\newcommand{\RESERVEDCOMMANDSourceCodeIndexFontSize}{\scriptsize}

\lstset
{
	%Font style for all the characters within a listing environment
	basicstyle = \RESERVEDCOMMANDSourceCodeIndexFontSize \ttfamily \color{black},
	stringstyle = \itshape\color{magenta},
	showstringspaces = false,
	commentstyle = \color{blue}\slshape,
	framexleftmargin = 1mm,

	keywordstyle = {[2]\bfseries\color{mbleu}},

	%The step between two line-numbers. If it is 1 each line will be numbered
	stepnumber = 1,

	%Underlines spaces within strings

	%Shows tabs withing strings adding particular underscores

\lstdefinestyle{block}
{
	frame = shadowbox,
	%Represents the spacing intended (default: fixed; flexible, fullflexible)
	columns = fullflexible,
	%Thickness of the edge of the box
	framerule = 0.1pt,
	%Separafion of the frame from the text.
	framesep = 3pt,
	%Position of line numbers
	numbers = left,
	%Color of the edge of the box
	rulecolor = \color{black},
	%Color of the inside of the box
	rulesepcolor = \color{gray!30},
}
