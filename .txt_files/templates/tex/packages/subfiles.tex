%subfiles{{{

%let handling multi-file projects.
\usepackage{subfiles}

%}}}

}}}
Examples {{{

1 {{{

./main.tex {{{

\documentclass{article}

\usepackage{subfiles}

\begin{document}
	\subfile{file_1.tex} 
	\subfile{file_2.tex} 
\end{document}

}}}

./file_1.tex {{{

\documentclass{subfiles}

\begin{document}
	File 1
\end{document}

}}}

./file_2.tex {{{

\documentclass{subfiles}

\begin{document}
	File 2
\end{document}

}}}

}}}

}}}
